// needed to declare what program is used in this case bash
#!/bin/bash

// if no input then return a string
if [ "$1" == "" ]
then
echo "You forgot the Ip address"
else
// ` is to the left of the 1
for ip in `seq 1 254`; do
// sends one ping greps the line with 64 bytes cuts it at the 4th " " and trims
// the : at the end. also pipes it into the second paramter as a file name.
ping -c 1 $1.$ip | grep "64 bytes" | cut -d " " -f 4 | tr -d ":" >> $2 &
done
fi
