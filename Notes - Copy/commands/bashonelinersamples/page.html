// goes through a list of IPs one line at a time does an nmap scan on port 80
// for each IP it creates a new file with nmap10.10.10.10 format.
// &= multithread

for ip in $(cat IPs.txt); do nmap -p 80 -T4 $ip -o nmap$ip & done
